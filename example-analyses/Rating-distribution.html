def get_dist_plot(contest, alg="KLUCB"):
    df = ccd.summary(contest)

    ax = df.score.hist(bins=20, grid=False)
    ax.set_xlim(1, 3)
    ax.set_xticks([1, 2, 3])
    ax.set_xticklabels(["Unfunny", "Somewhat funny", "Funny"])
    ax.set_title(f"Ratings for contest={contest}")
    ax.set_yscale("log")
    return df, ax
